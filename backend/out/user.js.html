const router = require('express').Router()
const { createUser, listUser, resetPasswordUser, updateVehicleUser } = require('../controllers/user.controller')

/**
 * Route serving list user.
 * @name get/list
 * @function
 * @param {string} path - Express path

 */

router.get('/list', listUser)

/**
 * Route serving create user.
 * @name post/add
 * @function

 */


router.post('/add', createUser)

/**
 * Route serving reset password user.
 * @name post/reset
 * @function

router.post('/reset', resetPasswordUser)
